'=======================Команды линукс========================='

ls - показывает содержимое папки файлы и другие
ls название директории (py33) - показывает содержимое
ls -a = показывает скрытые файлы и папки 
ls -l = доп информация о файлах и папках
ls -la = показывает скрытые файлы и директории с подробной информацией


cd ~ - перейти в домашнюю директорию 
cd - переход в домашнюю директорию
cd test - переход в папку test, если нет такой директории то выйдет ошибка
cd .. - переход на одну директорию выше 
cd ../.. - переход на две директории выше
cd - = cntr + z

mkdir - создаёт папку в текущей директории
mkdir test/test2 - создаёт папку test2 в внутри папки test если папка существует  
mkdir test test2 - создаё сразу две папки
touch file.txt - создаёт файл  (txt тип файла)
touch file1.txt file2.txt - создаёт сразу два файла
rm - удалить файл 
rmdir - удалить пустую директорию 
rm -rf = принужденное удаление директорий и файлов с содержанием и пустых. Рекурсивное удаление (полное удаление)
mv test.txt test - перемещение в директорию test 
mv - а также можно переименовать (test.tst на test1.txt)
pwd - показывает абсолютнвй путь (начиная с корневой попки до текущей директории)
history - показывает все команды которые мы вводили в данный терминал
history 10 - показывает 10 последних введеных команд из истории команд 
history | grep test = показывает команды из истории команд с словом txt
cat - выводит содержимого файла 
sudo - (super user do)- права суперюзера
apt - системный установщик пакетов (ubuntu) 
echo - 
du - показывает размер файла
